  230.             $listEntry = Jadva_File::getInstanceFor($path);
  231.  
  232.             if( (NULL !== $filter) && !$filter->filter($listEntry) ) {
  233.                 continue;
  234.             }
  235.  
  236.             $this->_list[] = $listEntry;
  237.             $this->_listCount++;
  238.         }
  239.     }
