  35. /**
  36.  * This class can be used to install database scripts.
  37.  *
  38.  * The advantage of this class is that you can build up your scripts in purely SQL fashion, and then execute them. You
  39.  * can state for each script the names and versions of the other scripts that are used. The filenames must be
  40.  *
  41.  * <scriptName>.<scriptVersion>.<databaseType>
  42.  *
  43.  * So for the Jadva_Installer_Database_Mysqli, the filenames could be as follows:
  44.  * <ul>
  45.  *   <li>users.1.mysql</li>
  46.  *   <li>users.2.mysql</li>
  47.  *   <li>user_logs.1.mysql</li>
  48.  * </ul>
  49.  *
  50.  * Note that each version will automatically depend on the versions before them, allowing you to update the tables etc
  51.  * you created in the first version. The version numbers will be stored in the database, allowing you to update your
  52.  * database at a later point in time. Version numbers are required to start at 1, and increase with 1 at each version.
  53.  *
  54.  * Example:
  55.  * <code>
  56.  * $outputFormatter = new Jadva_Installer_OutputFormatter_Xhtml;
  57.  * $outputFormatter->setTitle('Database installer example');
  58.  * 
  59.  * $installer = new Jadva_Installer_Database_Mysqli;
  60.  * $installer->setOutputFormatter($outputFormatter)
  61.  *           ->setRestoreDirectory('./restore/')
  62.  *           ->setCredentials('db_user_name', 'db_user_pass', 'localhost')
  63.  *           ->setDatabaseName('db_name')
  64.  *           ->addDirectory('./forum/installationscripts/')
  65.  *           ->addDirectory('./blog/installationscripts/')
  66.  *           ->install();
  67.  * </code>
  68.  *
  69.  * This example will load the database scripts from './forum/installationscripts/' and './blog/installationscripts/',
  70.  * make a backup of the database in './restore/', and execute the scripts against the database with 'db_name', after
  71.  * logging in to 'localhost' with 'db_user_name' and 'db_user_pass'. If the installation goes wrong, the backup will be
  72.  * restored.
  73.  *
  74.  * Requirements for the Jadva_Installer_Database_Mysqli are annotated in the .mysql files as follows:
  75.  * <ul>
  76.  *   <li>Start a line with a single-line comment symbol (-- or #)</li>
  77.  *   <li>Then state 'REQUIRES:' (with no space between "REQUIRES" and the ":")</li>
  78.  *   <li>Then state the name the script depends on</li>
  79.  *   <li>Then state a comma (",")</li>
  80.  *   <li>Then state the version number the script depends on</li>
  81.  * </ul>
  82.  *
  83.  * Support for other DBMSs is not written yet; there is a {@link Jadva_Installer_Database_Zend} class which connects
  84.  * using the Zend_Db_Adapter you specifiy, however, this has been untested, and creating a backup might not even be
  85.  * possible while using Zend. Therefore, the above stated requirements for MySQLi should be generalised to other DBMSs,
  86.  * but this is not written yet.
  87.  *
  88.  * @todo Add a 'done' message to the log, it's a bit unclear ATM whether it finished or not.
  89.  *
  90.  * @category   JAdVA
  91.  * @package    Jadva_Installer
  92.  * @subpackage Jadva_Installer_Database
  93.  * @copyright  Copyright (c) 2008 Ja`Achan da`Variso (http://www.JaAchan.com/)
  94.  * @license    http://www.JaAchan.com/software/LICENSE.txt
  95.  */

  193.     /**
  194.      * Installs the database scripts
  195.      *
  196.      * This function loads the installation script from the added directories, connects to the database and tries
  197.      * to lock it. Then it creates a restore point, orders the scripts, checks which versions are installed and
  198.      * updates the database. Finally, the database will be unlocked.
  199.      *
  200.      * If something goes wrong, it will try to restore the restore point. Should that fail, it will output the
  201.      * location of the restore point, so the user can restore the database themselves
  202.      *
  203.      * @return void 
  204.      */
  205.     public function install()
  206.     {
  207.         if( NULL === $this->_dbType ) {
  208.             throw new Exception('Database type unset');
  209.         }
  210.  
  211.         if( NULL === $this->_outputFormatter ) {
  212.             throw new Exception('Output formatter unset');
  213.         }
  214.  
  215.         if( NULL === $this->_restoreDirectory ) {
  216.             throw new Exception('Restore directory unset');
  217.         }
  218.  
  219.         if( NULL === $this->_credentialsUser ) {
  220.             throw new Exception('Credentials unset');
  221.         }
  222.  
  223.         if( NULL === $this->_databaseName ) {
  224.             throw new Exception('Database name unset');
  225.         }
  226.  
  227.         if( count($this->_directoryList) === 0 ) {
  228.             throw new Exception('No directories added');
  229.         }
  230.  
  231.         $this->_outputFormatter->outputStart();
  232.  
  233.         try{
  234.             $this->_outputFormatter->outputInfo('Reading the directories');
  235.             foreach($this->_directoryList as $directory) {
  236.                 $this->_loadFilesFromDirectory($directory);
  237.             }
  238.  
  239.             $this->_outputFormatter->outputInfo('Connecting to the database');
  240.             $this->_connectToDatabase();
  241.  
  242.             $this->_outputFormatter->outputInfo('Locking the database');
  243.             $this->_lockDatabase();
  244.  
  245.             $this->_outputFormatter->outputInfo('Creating a restore point');
  246.             $this->_createRestorePoint();
  247.  
  248.             $this->_outputFormatter->outputInfo('Retrieving the list of files');
  249.             $list = $this->_nodeLists[$this->_dbType]->getNodesAsTopologicalSort();
  250.             $this->_outputFormatter->outputSuccess('File list updated and sorted');
  251.  
  252.             $this->_outputFormatter->outputInfo('Checking version table');
  253.             $this->_checkVersionTable();
  254.  
  255.             $this->_outputFormatter->outputInfo('Retrieving current versions');
  256.             $oldVersionList     = $this->_getVersions();
  257.             $versionListUpdates = array();
  258.  
  259.             $this->_outputFormatter->outputInfo('Updating the database');
  260.             $updateCount = 0;
  261.             foreach($list as $node) {
  262.                 $currentVersion = intval(@$oldVersionList[$node->scriptName]);
  263.                 if( $currentVersion < $node->scriptVersion ) {
  264.                     $this->_outputFormatter->outputInfo('Updating script "' . $node->scriptName . '" to version ' . $node->scriptVersion);
  265.                     if( NULL === $node->content ) {
  266.                         /** @see Jadva_Installer_Database_Exception */
  267.                         require_once 'Jadva/Installer/Database/Exception.php';
  268.                         throw new Jadva_Installer_Database_Exception('Missing content for file "' . $node->filename . '"');
  269.                     }
  270.  
  271.                     $this->_executeDatabaseScript($node->content);
  272.  
  273.                     $updateCount++;
  274.                     $oldVersionList[$node->scriptName] = $node->scriptVersion;
  275.                     $versionListUpdates[$node->scriptName] = $node->scriptVersion;
  276.                     $this->_outputFormatter->outputSuccess('Update successful');
  277.                 }
  278.             }
  279.  
  280.             if( $updateCount > 0 ) {
  281.                 $this->_outputFormatter->outputSuccess('Updated ' . $updateCount . ' scripts');
  282.                 $this->_storeVersions($versionListUpdates);
  283.             } else {
  284.                 $this->_outputFormatter->outputSuccess('Database is up to date');
  285.             }
  286.         } catch(Jadva_Installer_Database_Exception $e) {
  287.             if( NULL === $this->_restorePointLocation ) {
  288.                 $this->_outputFormatter->outputCrit('An error has occurred before or during the creating of the restore point');
  289.                 $this->_outputFormatter->outputErr($e->getMessage());
  290.             } else {
  291.                 $this->_outputFormatter->outputErr ('An error has occurred: ' . $e->getMessage());
  292.                 $this->_outputFormatter->outputInfo('Restoring restore point');
  293.                 try {
  294.                     $this->_restoreRestorePoint();
  295.                 } catch(Jadva_Installer_Database_Exception $e2) {
  296.                     $this->_outputFormatter->outputErr ('An error has occurred while restoring the restore point: ' . $e->getMessage());
  297.                 }
  298.             }
  299.         }
  300.  
  301.         if( $this->_databaseIsLocked() ) {
  302.             $this->_outputFormatter->outputInfo('Unlocking the database');
  303.             $this->_unlockDatabase();
  304.         }
  305.  
  306.         $this->_disconnectFromDatabase();
  307.  
  308.         $this->_outputFormatter->outputEnd();
  309.     }

  377.     /**
  378.      * Checks a directory for existance
  379.      *
  380.      * @param  string   $directory  The directory to check
  381.      * @param  boolean  $writable   If TRUE, this also checks whether the directory is writable
  382.      *
  383.      * @return  The cleaned up directory URI
  384.      */
  385.     protected function _checkDirectory($directory, $writable = FALSE)
  386.     {
  387.         $directory = (string) $directory;
  388.         $directory = str_replace(array('\\', '/'), DIRECTORY_SEPARATOR, $directory);
  389.         if( substr($directory, -1) !== DIRECTORY_SEPARATOR ) {
  390.             $directory .= DIRECTORY_SEPARATOR;
  391.         }
  392.  
  393.         if(!is_dir($directory) ) {
  394.             throw new Exception('No such directory: "' . $directory . '"');
  395.         }
  396.  
  397.         if( $writable && !is_writeable($directory) ) {
  398.             throw new Exception('Cannot write to directory: "' . $directory . '"');
  399.         }
  400.  
  401.         return $directory;
  402.     }
  403.     //------------------------------------------------
  404.     /**
  405.      * Loads the database scripts from the given directory and its subdirectories
  406.      *
  407.      * @param  string  $directory  The directory from which to load the scripts
  408.      *
  409.      * @return  void 
  410.      */
  411.     protected function _loadFilesFromDirectory($directory)
  412.     {
  413.         $iterator = new DirectoryIterator($directory);
  414.         foreach($iterator as $file) {
  415.             if( $file->isDot() ) {
  416.                 continue;
  417.             }
  418.  
  419.             $filename = $file->getFileName();
  420.             if( '.' === $filename[0] ) {
  421.                 continue;
  422.             }
  423.  
  424.             $fullUri = $file->getPath() . DIRECTORY_SEPARATOR . $filename;
  425.             if( $file->isDir() ) {
  426.                 $this->_loadFilesFromDirectory($fullUri . DIRECTORY_SEPARATOR);
  427.                 return;
  428.             }
  429.  
  430.             $list = explode('.', $filename);
  431.
