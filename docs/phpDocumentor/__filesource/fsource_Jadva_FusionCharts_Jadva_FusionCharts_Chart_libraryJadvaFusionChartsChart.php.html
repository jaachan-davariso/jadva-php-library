  185.     public function setAutoColour($scheme = self::COLOUR_SCHEME_FUSION_CHARTS)

  203.             $this->_colourSchemeName = self::COLOUR_SCHEME_CUSTOM;
  204.         } elseif( array_key_exists($scheme, self::$_colourSchemes) ) {
  205.             $this->_colourSchemeName = $scheme;

  248.     /**
  249.      * Sets whether to apply our y-axis min and max fix
  250.      *
  251.      * FusionCharts doesn't take trend lines in account, and doesn't account for graphs with only 0 values. We try
  252.      * to fix that:
  253.      * <ul>
  254.      *   <li>If a value contains only 0 values, we set the max y-axis value to 5, ensuring that you don't get weird
  255.      *       NaN's</li>

  257.      * </ul>
  258.      *
  259.      * It's off by default. It doesn't work for stacked charts (yet).
  260.      *
  261.      * @param  boolean  $in_yaxisMinMaxFix  TRUE to apply our y-axis min and max fix, FALSE not to apply it.
  262.      *
  263.      * @todo  Fix it so this works for stacked charts as well.
  264.      * @return  Jadva_FusionCharts_Chart  Provides a fluent interface

  291.             $this->_colourSchemeName
  292.         &&
  293.             (self::SERIES_TYPE_MULTI_SERIES === $this->getSeriesType())
  294.         &&
  295.             !array_key_exists('color', $this->_datasetAttributeList[$this->getDataSetCount() - 1])

  346.     public function setDataSetCount($in_dataSetCount)
  347.     {
  348.         $dataSetCount = (int) $in_dataSetCount;

  350.         if( !$this->hasSeriesType() ) {
  351.             /** @see Jadva_FusionCharts_Exception */
  352.             require_once 'Jadva/FusionCharts/Exception.php';

  354.         }
  355.  
  356.         if( (self::SERIES_TYPE_SINGLE_SERIES == $this->getSeriesType()) && (1 !== $dataSetCount) ) {
  357.             /** @see Jadva_FusionCharts_Exception */

  360.         }
  361.  
  362.         if( $this->hasDataSetCount() ) {
  363.             $oldCount = $this->getDataSetCount();
  364.         } else {

  366.         }
  367.  
  368.         $this->_dataSetCount = $dataSetCount;
  369.  
  370.         for($it = $oldCount; $it < $dataSetCount; $it++) {

  372.             $this->_datasetAttributeList[$it] = array();
  373.         }
  374.  
  375.         for($it = $dataSetCount; $it < $oldCount; $it++) {
  376.             unset($this->_values[$it]);
  377.             unset($this->_datasetAttributeList[$it]);
  378.         }
  379.  
  380.         return $this;
  381.     }

  399.         }
  400.  
  401.         if( ($dataSetNumber < 0) || ($this->getDataSetCount() <= $dataSetNumber) ) {
  402.             /** @see Jadva_FusionCharts_Exception */
  403.             require_once 'Jadva/FusionCharts/Exception.php';
  404.             throw new Jadva_FusionCharts_Exception('Data set number must be between 0 and ' . ($this->getDataSetCount() - 1));
  405.         }
  406.  
  407.         $attributeList = array();
  408.         foreach($attribs as $key => $value) {
  409.             $key = strtolower(trim($key));
  410.  
  411.             $attributeList[$key] = $value;
  412.         }
  413.  
  414.         $this->_datasetAttributeList[$dataSetNumber] = $attributeList;
  415.  
  416.         return $this;
  417.     }

  598.     {

  601.             $value = NULL;

  643.     }

  765.     }
