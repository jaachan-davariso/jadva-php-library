  87.     {
  88.         $seriesType = (int) $in_seriesType;
  89.  
  90.         if( ($seriesType <= 0) || (6 <= $seriesType) ) {

  95.  
  96.         if( $this->hasGraphType() ) {
  97.             if( self::$_graphTypeToSeriesType[$this->getGraphType()] !== $seriesType ) {
  98.                 $this->_graphType = NULL;
  99.             }
  100.         } else {
  101.             $this->_graphType = self::$_seriesTypeToDefaultGraphType[$seriesType];

  104.         $this->_seriesType = $seriesType;
  105.  
  106.         if( (self::SERIES_TYPE_SINGLE_SERIES === $seriesType) && !$this->hasDataSetCount() ) {
  107.             $this->setDataSetCount(1);
  108.         }
  109.  
  110.         return $this;
  111.     }

  188.     public function setGraphType($in_type)
  189.     {
  190.         $type = (string) $in_type;
  191.  
  192.         if( !array_key_exists($type, self::$_graphTypeToSeriesType) ) {
  193.             /** @see Jadva_FusionCharts_Exception */
  194.             require_once 'Jadva/FusionCharts/Exception.php';
  195.             throw new Jadva_FusionCharts_Exception('No such graph type: "' . $in_type .'"');
  196.         }
  197.  
  198.         $graphTypeSeriesType = self::$_graphTypeToSeriesType[$type];
  199.         if( $this->hasSeriesType() ) {
  200.             if( $this->getSeriesType() !== $graphTypeSeriesType ) {
  201.                 /** @see Jadva_FusionCharts_Exception */
  202.                 require_once 'Jadva/FusionCharts/Exception.php';
  203.                 throw new Jadva_FusionCharts_Exception(
  204.                     'The series type for the new chart type ("' . $graphTypeSeriesType
  205.                       . '") does not match the one already set for this graph ("' . $this->getSeriesType()
  206.                       . '").'
  207.                 );
  208.             }
  209.         } else {
  210.             $this->setSeriesType($graphTypeSeriesType);
  211.         }
  212.  
  213.         $this->_graphType = $in_type;
  214.  
  215.         return $this;
  216.     }

  251.     /**
  252.      * Adds a value to a dataset
  253.      *
  254.      * @param  integer  $in_dataSetNumber  The number of the data set
  255.      * @param  float    $in_value          The value to add
  256.      * @param  array    $attribs           (OPTIONAL) Additonal attributes for the value
  257.      *
  258.      * @return  Jadva_FusionCharts_Chart  Provides a fluent interface
  259.      */
  260.     public function addValue($in_dataSetNumber, $in_value, array $attribs = array())
  261.     {
  262.         $dataSetNumber = (int) $in_dataSetNumber;
  263.         $value         = (float) $in_value;
  264.  
  265.  
  266.         if( !$this->hasDataSetCount() ) {
  267.             /** @see Jadva_FusionCharts_Exception */
  268.             require_once 'Jadva/FusionCharts/Exception.php';
  269.             throw new Jadva_FusionCharts_Exception('Set the data count first');
  270.         }
  271.         if( ($dataSetNumber < 0) || ($this->getDataSetCount() <= $dataSetNumber) ) {
  272.             /** @see Jadva_FusionCharts_Exception */
  273.             require_once 'Jadva/FusionCharts/Exception.php';
  274.             throw new Jadva_FusionCharts_Exception('Data set number must be between 0 and ' . $this->getDataSetCount() - 1);
  275.         }
  276.  
  277.         $options = array(
  278.             'value' => $value,
  279.         );
  280.  
  281.         if( self::SERIES_TYPE_SINGLE_SERIES === $this->getSeriesType() ) {
  282.             $allowedAttributes = &self::$_allowdSetAttributesSingleSeries;
  283.         } else {
  284.             $allowedAttributes = &self::$_allowdSetAttributesOther;
  285.         }
  286.  
  287.         foreach($attribs as $key => $attribValue) {
  288.             if( in_array($key, $allowedAttributes) ) {
  289.                 $options[$key] = $attribValue;
  290.             }
  291.         }
  292.  
  293.         $this->_values[$dataSetNumber][] = $options;
  294.  
  295.         if( NULL === $this->_maxValue ) {
  296.             $this->_maxValue = $value;
  297.         } else {
  298.             $this->_maxValue = max($this->_maxValue, $value);
  299.         }
  300.  
  301.         return $this;
  302.     }

  304.     /**
  305.      * Generates an XML document of this chart

  309.     public function toXml()
  310.     {
  311.         $document = new DOMDocument;
  312.  
  313.         $graph = $document->createElement('graph');
  314.  
  315.         foreach($this->_graphAttributes as $key => $value ) {
  316.             $graph->setAttribute($key, $value);

  318.  
  319.         if( !array_key_exists('yAxisMaxValue', $this->_graphAttributes) ) {
  320.             $graph->setAttribute('yAxisMaxValue', $this->_maxValue * 1.1);
  321.         }
  322.  
  323.         $dataSetCount = $this->getDataSetCount();
  324.         for($dataSetNumber = 0; $dataSetNumber < $dataSetCount; $dataSetNumber++) {
  325.             if( 1 < $dataSetCount ) {
  326.                 $dataSet = $document->createElement('dataset');
  327.             } else {
  328.                 $dataSet = $graph;
  329.             }
  330.  
  331.             foreach($this->_values[$dataSetNumber] as $valueInfo) {
  332.                 $set = $document->createElement('set');
  333.                 $set->setAttribute('value', $valueInfo['value']);
  334.  
  335.                 if( 1 == $dataSetCount ) {
  336.                     $name = $valueInfo['name'];
  337.                     if( !empty($name) ) {
  338.                         $set->setAttribute('name', $name);
  339.                     }
  340.                 }
  341.  
  342.                 $dataSet->appendChild($set);
  343.             }
  344.  
  345.             if( 1 < $dataSetCount ) {
  346.                 $document->appendChild($dataSet);
  347.             }
  348.         }
  349.  
  350.         $document->appendChild($graph);
  351.  
  352.         return $document->saveXML();
  353.     }
