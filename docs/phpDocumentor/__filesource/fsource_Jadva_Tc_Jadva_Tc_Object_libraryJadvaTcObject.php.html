  30. /**
  31.  * This class represenst an object
  32.  *
  33.  * A lot of the TC objects link to each other, leading to circular references. This class tries to help, by creating
  34.  * a sort of 'state' for an object. Also adds a few helper functions, like the automatical __get for getVarname
  35.  * functions.
  36.  *
  37.  * @category   JAdVA
  38.  * @package    Jadva_Tc
  39.  * @subpackage Jadva_Tc_Object
  40.  * @copyright  Copyright (c) 2008 Ja`Achan da`Variso (http://www.JaAchan.com/)
  41.  * @license    http://www.JaAchan.com/software/LICENSE.txt
  42.  */
  43. abstract class Jadva_Tc_Object
  44. {
  45.     //------------------------------------------------
  46.     /**
  47.      * An object in this state is very much alive
  48.      */
  49.     const STATE_ALIVE  = 'stateAlive';
  50.     //------------------------------------------------
  51.     /**
  52.      * An object in this state has been called to be deleted, but hasn't finished the deletion yet
  53.      */
  54.     const STATE_DIEING = 'stateDieing';
  55.     //------------------------------------------------
  56.     /**
  57.      * An object in this state has been called to be deleted, and has finished the deletion. Should not be used
  58.      * anymore, so the garbage collector can come pick it up
  59.      */
  60.     const STATE_DEAD   = 'stateDead';
  61.     //------------------------------------------------
  62.     /**
  63.      * Default constructor.
  64.      *
  65.      * Generates an class-unique identity
  66.      */
  67.     public function __construct()
  68.     {
  69.         static $identity_list = array();
  70.  
  71.         $class = get_class($this);
  72.         if( !array_key_exists($class, $identity_list) ) {
  73.             $identity_list[$class] = 0;
  74.         }
  75.  
  76.         $this->_identity = $identity_list[$class];
  77.         $identity_list[$class]++;
  78.  
  79.         $this->_aliveState = self::STATE_ALIVE;
  80.     }
  81.     //------------------------------------------------
  82.     /**
  83.      * Default destructor
  84.      *
  85.      * Calls upon {@link delete()} if still alive
  86.      */
  87.     public function __destruct()
  88.     {
  89.         if( self::STATE_DEAD !== $this->_aliveState ) {
  90.             $this->delete();
  91.         }
  92.     }
  93.     //------------------------------------------------
  94.     /**
  95.      * Returns the identity of this object
  96.      *
  97.      * @return  The identity of this object
  98.      */
  99.     public function getIdentity()
  100.     {
  101.         return $this->_identity;
  102.     }
  103.     //------------------------------------------------
  104.     /**
  105.      * Deletes this object, removing any references to other objects
  106.      */
  107.     public function delete()
  108.     {
  109.         if( self::STATE_DEAD === $this->_aliveState ) {
  110.             return;
  111.         }
  112.  
  113.         if( self::STATE_DIEING === $this->_aliveState ) {
  114.             return;
  115.         }
  116.         $this->_aliveState = self::STATE_DIEING;
  117.  
  118.         $this->_innerDelete();
  119.  
  120.         $this->_aliveState = self::STATE_DEAD;
  121.     }
  122.     //------------------------------------------------
  123.     /**
  124.      * Calls upon the set{$varName} function if exists, triggers an error otherwise
  125.      *
  126.      * @param  string  $varName   The name of the variable that was tried to set
  127.      * @param  mixed   $newValue  The value that it was tried to set to
  128.      *
  129.      * @return mixed 
  130.      */
  131.     public function __set($varName, $newValue) {
  132.         $function = 'set' . ucfirst($varName);
  133.         if( method_exists($this, $function) ) {
  134.             return $this->$function($newValue);
  135.         }
  136.  
  137.         trigger_error('Class ' . get_class($this) . ' has no property called "' . $varName .'"', E_USER_WARNING);
  138.     }
  139.     //------------------------------------------------
  140.     /**
  141.      * Calls upon the get{$varName} function if exists, triggers an error otherwise
  142.      *
  143.      * @param  string  $varName   The name of the variable that was tried to get
  144.      *
  145.      * @return mixed 
  146.      */
  147.     public function __get($varName) {
  148.         $function = 'get' . ucfirst($varName);
  149.         if( method_exists($this, $function) ) {
  150.             return $this->$function();
  151.         }
  152.  
  153.         trigger_error('Class ' . get_class($this) . ' has no property called "' . $varName .'"', E_USER_WARNING);
  154.     }
  155.     //------------------------------------------------
  156.     /**
  157.      * Throws an exception if the object is already dead
  158.      *
  159.      * @throws  Jadva_Tc_Exception  If the object is already dead
  160.      */
  161.     protected function _bailIfDead()
  162.     {
  163.         if( self::STATE_DEAD === $this->_aliveState ) {
  164.             require_once 'Jadva/Tc/Exception.php';
  165.             throw new Jadva_Tc_Exception('This object is dead and can therefore not be accessed anymore');
  166.         }
  167.     }
  168.     //------------------------------------------------
  169.     /**
  170.      * Gets called when the object is called to delete
  171.      *
  172.      * @return  void 
  173.      */
  174.     abstract protected function _innerDelete();
  175.     //------------------------------------------------
  176.     /**
  177.      * Contains the class-unique identity of this object
  178.      *
  179.      * @var  integer 
  180.      */
  181.     private $_identity;
  182.     //------------------------------------------------
  183.     /**
  184.      * Contains the liveness-state of this object
  185.      *
  186.      * @var  string 
  187.      */
  188.     private $_aliveState;
  189.     //------------------------------------------------
  190. }
