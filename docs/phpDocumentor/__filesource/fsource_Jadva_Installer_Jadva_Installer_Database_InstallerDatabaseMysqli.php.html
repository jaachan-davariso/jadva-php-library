  328.     /**
  329.      * Implements Jadva_Installer_Database_Abstract::_parseDatabaseScript
  330.      *
  331.      * In general, we try to accept the same behaviour as the mysql command line tool does.
  332.      *
  333.      * We need to feed the queries, one at a time. And we want to be able to deal with DELIMITER clauses. Hence, a
  334.      * rather large parser was hidden away in here
  335.      *
  336.      * When you use "#" or "--", the mysql program acts odd. Hence, we assume people don't feed us that kind of
  337.      * scripts either
  338.      *
  339.      * Assumed is that the ";" is used as a delimiter at the beginning of EACH file
  340.      *
  341.      * MySQL uses /*! as a special kind of extension. Therefore, we don't remove /* comments. As far as we can see,
  342.      * this does not interfere with the "one query at a time", since you cannot hide entire queries with this from
  343.      * other DBMS systems.
  344.      */
  345.     protected function _parseDatabaseScript($scriptContents)
  346.     {
  347.         $parserState = self::PARSER_STATE_OUT;
  348.  
  349.         $delimiter   = ';'; $delimiterLength = strlen($delimiter);
  350.  
  351.         $queryList           = array();
  352.         $oneLineCommentsList = array();
  353.  
  354.         $curIndex = 0; $scriptLength = strlen($scriptContents);
  355.  
  356.         $curText           = '';
  357.         $curOneLineComment = '';
  358.  
  359.         while( $curIndex < $scriptLength ) {
  360.             $char = $scriptContents[$curIndex];
  361.  
  362.             switch($parserState) {
  363.             case self::PARSER_STATE_OUT:
  364.                 if( substr($scriptContents, $curIndex, $delimiterLength) === $delimiter ) {
  365.                     $queryList[] = $curText;
  366.                     $curText = '';
  367.                     $curIndex += $delimiterLength;
  368.                     continue;
  369.                 }
  370.  
  371.                 switch($char) {
  372.                 case '-':
  373.                     $parserState = self::PARSER_STATE_DASH;
  374.                     break;
  375.                 case "'":
  376.                     $parserState = self::PARSER_STATE_QUOTE_ONE;
  377.                     $curText .= $char;
  378.                     break;
  379.                 case '"':
  380.                     $parserState = self::PARSER_STATE_QUOTE_TWO;
  381.                     $curText .= $char;
  382.                     break;
  383.                 case '#':
  384.                     $parserState = self::PARSER_STATE_COMMENT_ONELINE;
  385.                     $curText .= $char;
  386.                     break;
  387.                 case "/":
  388.                     $parserState = self::PARSER_STATE_COMMENT_MULTI_START;
  389.                     break;
  390.                 default:
  391.                     //The proper way of doing this would be to introduce 9 extra states
  392.                     // But this is much easier
  393.                     if( 'd' === strtolower($char) ) { //Quick test
  394.                         //Could be 'delimiter'
  395.                         if( 'delimiter' === strtolower(substr($scriptContents, $curIndex, 9)) ) {
  396.                             $tmpPos = $curIndex + 9;
  397.                             while( ($tmpPos < $scriptLength) && (' '  === $scriptContents[$tmpPos]) ) { $tmpPos++; }
  398.  
  399.                             $newLinePos = strpos($scriptContents, "\n", $tmpPos);
  400.  
  401.                             $delimiter       = trim(substr($scriptContents, $tmpPos, $newLinePos - $tmpPos));
  402.                             $delimiterLength = strlen($delimiter);
  403.  
  404.  
  405.                             $curIndex = $newLinePos; //It gets added one for the \n later on
  406.                             $parserState = self::PARSER_STATE_OUT;
  407.                         } else {
  408.                             $parserState = self::PARSER_STATE_OUT;
  409.                             $curText .= $char;
  410.                         }
  411.                     } else {
  412.                         $parserState = self::PARSER_STATE_OUT;
  413.                         $curText .= $char;
  414.                     }
  415.                     break;
  416.                 }
  417.                 break;
  418.  
  419.             case self::PARSER_STATE_DASH:
  420.                 switch($char) {
  421.                 case '-':
  422.                     $parserState = self::PARSER_STATE_COMMENT_ONELINE;
  423.                     break;
  424.                 default:
  425.                     $parserState = self::PARSER_STATE_OUT;
  426.                     $curText .= '-'; //Add a dash for the one that got us into this state
