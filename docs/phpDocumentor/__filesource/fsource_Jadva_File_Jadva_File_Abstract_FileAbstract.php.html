  53.     /**
  54.      * Returns an instance for the given path, existing or not

  60.     public static function getInstanceFor($in_path)
  61.     {
  62.         if( $in_path instanceof self ) {
  63.             return $in_path;
  64.         }
  65.  
  66.         $path = self::cleanPath($in_path);
  67.  
  68.         if( is_dir($path) ) {
  69.             $className = 'Jadva_File_Directory';
  70.             $path = rtrim($path, DIRECTORY_SEPARATOR) . DIRECTORY_SEPARATOR;
  71.         } elseif( file_exists($path) ) {
  72.             $className = 'Jadva_File';
  73.             $path = rtrim($path, DIRECTORY_SEPARATOR);
  74.         } else {
  75.             //No existing file or directory, guess which one it should be base on the convention
  76.             // that directories are always passed with a DIRECTORY_SEPARATOR at the end
  77.             if( DIRECTORY_SEPARATOR == substr($path, -1) ) {
  78.                 $className = 'Jadva_File_Directory';
  79.             } else {
  80.                 $className = 'Jadva_File';
  81.             }
  82.         }
  83.  
  84.         if( !class_exists($className, FALSE) ) {
  85.             require_once str_replace('_', DIRECTORY_SEPARATOR, $className) . '.php';
  86.         }
  87.  
  88.         return new $className($path);
  89.     }
  90.     //------------------------------------------------
  91.     /**
  92.      * Verifies whether a file exists, and throws an exception if it doesn't
  93.      *
  94.      * @param  string   $in_path   The path to verifiy
  95.      * @param  integer  $in_flags  Which properties of the file to test
  96.      *
  97.      * @return  Jadva_File  The file on the given path
  98.      */
  99.     public static function verifyExistance($in_path, $in_flags = 5)
  100.     {
  101.         $file = self::getInstanceFor($in_path);
  102.  
  103.         if( !$file->exists() ) {
  104.             /** @see Jadva_File_Directory_Exception */
  105.             require_once 'Jadva/File/Directory/Exception.php';
  106.             throw new Jadva_File_Directory_Exception('The given directory ("' . $in_path . '") does not exist.');
  107.         }
  108.  
  109.         $flags = (int) $in_flags;
  110.  
  111.         if( ($flags & self::FLAG_IS_READABLE) && !$file->isReadable() ) {
  112.             /** @see Jadva_File_Directory_Exception */
  113.             require_once 'Jadva/File/Directory/Exception.php';
  114.             throw new Jadva_File_Directory_Exception('The given directory ("' . $in_path . '") is not readable.');
  115.         }
  116.  
  117.         if( ($flags & self::FLAG_IS_WRITABLE) && !$file->isWritable() ) {
  118.             /** @see Jadva_File_Directory_Exception */
  119.             require_once 'Jadva/File/Directory/Exception.php';
  120.             throw new Jadva_File_Directory_Exception('The given directory ("' . $in_path . '") is not writable.');
  121.         }
  122.  
  123.         if( ($flags & self::FLAG_IS_EXECUTABLE) && !$file->isExecutable() ) {
  124.             /** @see Jadva_File_Directory_Exception */
  125.             require_once 'Jadva/File/Directory/Exception.php';
  126.             throw new Jadva_File_Directory_Exception('The given directory ("' . $in_path . '") is not executable.');
  127.         }
  128.  
  129.         return $file;
  130.     }
