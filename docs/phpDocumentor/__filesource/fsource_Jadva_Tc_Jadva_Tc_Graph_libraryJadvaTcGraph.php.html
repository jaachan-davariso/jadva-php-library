  48. class Jadva_Tc_Graph extends Jadva_Tc_Object
  49. {
  50.     //------------------------------------------------
  51.     /**
  52.      * Default constructor
  53.      *
  54.      * @return  void 
  55.      */
  56.     public function __construct()
  57.     {
  58.         parent::__construct();
  59.  
  60.         $this->_nodeList = array();
  61.         $this->_edgeList = array();
  62.  
  63.         if( !class_exists($this->_nodeClass, TRUE) ) {
  64.             require_once 'Jadva/Tc/Graph/Exception.php';
  65.             throw new Jadva_Tc_Graph_Exception('Undefined node class "' . $this->_nodeClass . '"');
  66.         }
  67.  
  68.         if( ('Jadva_Tc_Node' !== $this->_nodeClass ) && !is_subclass_of($this->_nodeClass, 'Jadva_Tc_Node') ) {
  69.             require_once 'Jadva/Tc/Graph/Exception.php';
  70.             throw new Jadva_Tc_Graph_Exception('Node class must inherit from Jadva_Tc_Node');
  71.         }
  72.  
  73.         if( !class_exists($this->_edgeClass, TRUE) ) {
  74.             require_once 'Jadva/Tc/Graph/Exception.php';
  75.             throw new Jadva_Tc_Graph_Exception('Undefined edge class "' . $this->_edgeClass . '"');
  76.         }
  77.  
  78.         if( ('Jadva_Tc_Edge' !== $this->_edgeClass ) && !is_subclass_of($this->_edgeClass, 'Jadva_Tc_Edge') ) {
  79.             require_once 'Jadva/Tc/Graph/Exception.php';
  80.             throw new Jadva_Tc_Graph_Exception('Edge class must inherit from Jadva_Tc_Edge');
  81.         }
  82.  
  83.         $this->_dfsFinished        = TRUE;  //No nodes
  84.         $this->_graphContainsCycle = FALSE; //No nodes
  85.     }
  86.     //------------------------------------------------
  87.     /**
  88.      * Adds a node to the graph
  89.      *
  90.      * @param  array  $options  The options (ignored in this graph, can be used in descendants)
  91.      *
  92.      * @return  Jadva_Tc_Node  The new node that wast just added
  93.      */
  94.     public function addNode(array $options = array())
  95.     {
  96.         $newNode = new $this->_nodeClass($this);
  97.         $this->_nodeList[] = $newNode;
  98.         $this->_dfsFinished = FALSE;
  99.         return $newNode;
  100.     }
  101.     //------------------------------------------------
  102.     /**
  103.      * Adds an edge to the graph
  104.      *
  105.      * @param  Jadva_Tc_Node  $nodeOne  The node that the new edge starts from
  106.      * @param  Jadva_Tc_Node  $nodeTwo  The node that the new edge ends in
  107.      * @param  array          $options  The options (ignored in this graph, can be used in descendants)
  108.      *
  109.      * @return  Jadva_Tc_Edge  The new edge that wast just added
  110.      */
  111.     public function addEdge(Jadva_Tc_Node $nodeOne, Jadva_Tc_Node $nodeTwo, array $options = array())
  112.     {
  113.         $newEdge = new $this->_edgeClass($this, $nodeOne, $nodeTwo, FALSE);
  114.         $this->_edgeList[] = $newEdge;
  115.         $this->_dfsFinished = FALSE;
  116.         return $newEdge;
  117.     }
  118.     //------------------------------------------------
  119.     /**
  120.      * Removes an edge from the graph
  121.      *
  122.      * @param  Jadva_Tc_Edge  $edge  The edge to remove
  123.      *
  124.      * @return  void 
  125.      */
  126.     public function removeEdge(Jadva_Tc_Edge $edge)
  127.     {
  128.         if( $edge->getParent() !== $this ) {
  129.             require_once 'Jadva/Tc/Exception.php';
  130.             throw new Jadva_Tc_Exception('Parent mismatch');
  131.         }
  132.  
  133.         $edge->delete();
  134.         $this->_dfsFinished = FALSE;
  135.     }
  136.     //------------------------------------------------
  137.     /**
  138.      * Returns TRUE if this graph contains cycles, FALSE otherwise
  139.      *
  140.      * @return  boolean  TRUE if this graph contains cycles, FALSE otherwise
  141.      */
  142.     public function containsCycle()
  143.     {
  144.         $this->_dfs();
  145.  
  146.         return $this->_graphContainsCycle;
  147.     }
  148.     //------------------------------------------------
  149.     /**
  150.      * Returns clones of the nodes in this graph, sorted topological
  151.      *
  152.      * @return  array  The nodes
  153.      */
  154.     public function getNodesAsTopologicalSort()
  155.     {
  156.         $this->_dfs();
  157.  
  158.         if( $this->containsCycle() ) {
  159.             require_once 'Jadva/Tc/Graph/Exception.php';
  160.             throw new Jadva_Tc_Graph_Exception('Cannot retrieve topological sort for cyclic graph');
  161.         }
  162.  
  163.         $return = array();
  164.         foreach($this->_nodeList as $node) {
  165.             $return[ $node->getIdentity() ] = clone $node;
  166.         }
  167.  
  168.         uasort($return, create_function('$a, $b', 'return $b->finishingTime < $a->finishingTime;'));
  169.  
  170.         return $return;
  171.     }
  172.     //------------------------------------------------
  173.     /**
  174.      * Contains all the nodes in this graph
  175.      *
  176.      * @var  array 
  177.      */
  178.     protected $_nodeList;
  179.     //------------------------------------------------
  180.     /**
  181.      * Contains all the edge in this graph
  182.      *
  183.      * @var  array 
  184.      */
  185.     protected $_edgeList;
  186.     //------------------------------------------------
  187.     /**
  188.      * When $this->_dfsFinished, this variable contains TRUE if this graph contains cycles, FALSE otherwise
  189.      *
  190.      * @var  array 
  191.      */
  192.     protected $_graphContainsCycle;
  193.     //------------------------------------------------
  194.     /**
  195.      * Contains whether a DFS was finished for the current nodes/edges
  196.      *
  197.      * @var  boolean 
  198.      */
  199.     protected $_dfsFinished;
  200.     //------------------------------------------------
  201.     /**
  202.      * Contains the clock for the DFS function
  203.      *
  204.      * @var  boolean 
  205.      */
  206.     protected $_dfsTime;
  207.     //------------------------------------------------
  208.     /**
  209.      * Contains the class used for creating the added nodes
  210.      * @var  string 
  211.      */
  212.     protected $_nodeClass = 'Jadva_Tc_Node';
  213.     //------------------------------------------------
  214.     /**
  215.      * Contains the class used for creating the added edges
  216.      * @var  string 
  217.      */
  218.     protected $_edgeClass = 'Jadva_Tc_Edge';
  219.     //------------------------------------------------
  220.     /**
  221.      * Does a Depth First Search on the current graph
  222.      *
  223.      * @todo  Add support for $callbacks
  224.      *
  225.      * @param  array  $callbacks  Not implemented yet. Should handle callbacks for events in the DFS
  226.      *
  227.      * @return void 
  228.      */
  229.     protected function _dfs(array $callbacks = array())
  230.     {
  231.         if( $this->_dfsFinished ) {
  232.             return;
  233.         }
  234.  
  235.         $this->_graphContainsCycle = FALSE;
  236.  
  237.         //Set the colour to white (undiscovered) and clear other properties we'll set later
  238.         foreach($this->_nodeList as $node) {
  239.             $node->colour        = Jadva_Tc_Node::COLOUR_WHITE;
  240.             $node->predecessor   = NULL;
  241.             $node->discoveryTime = NULL;
  242.             $node->finishingTime = NULL;
  243.         }
  244.  
  245.         //Start the clock
  246.         $this->_dfsTime = 0;
  247.  
  248.         //Visit all nodes
  249.         foreach($this->_nodeList as $node) {
  250.             //But only if they're undiscovered
  251.             if( Jadva_Tc_Node::COLOUR_WHITE === $node->colour ) {
  252.                 $this->_dfsNodeVisit($node);
  253.             }
  254.         }
  255.  
  256.         $this->_dfsFinished = TRUE;
  257.     }
  258.     //------------------------------------------------
  259.     /**
  260.      * Visits a node in a DFS
  261.      *
  262.      * @todo  Add support for $callbacks
  263.      *
  264.      * @param  Jadva_Tc_Node  $nodeU      The node to visit
  265.      * @param  array          $callbacks  Not implemented yet. Should handle callbacks for events in the DFS
  266.      *
  267.      * @return void 
  268.      */
  269.     protected function _dfsNodeVisit(Jadva_Tc_Node $nodeU, array $callbacks = array())
  270.     {
  271.         //Set the colour to grey (discovered), increase the time and set the discovery time
  272.         $nodeU->colour        = Jadva_Tc_Node::COLOUR_GREY;
  273.         $nodeU->discoveryTime = ++$this->_dfsTime;
  274.  
  275.         //Visit all the nodes reachable from this node
  276.         foreach($nodeU->getOutgoingEdges() as $edge) {
  277.             $nodeV = $edge->getNodeTo();
  278.  
  279.             //If the node is grey (discovered), this edge is a Back Edge
  280.             if( Jadva_Tc_Node::COLOUR_GREY === $nodeV->colour ) {
  281.                 $this->_graphContainsCycle = TRUE;
  282.             }
  283.  
  284.             //If the node is white (undiscovered), visit it
  285.             if( Jadva_Tc_Node::COLOUR_WHITE === $nodeV->colour ) {
  286.                 $nodeV->predecessor = $nodeU;
  287.                 $this->_dfsNodeVisit($nodeV);
  288.             }
  289.         }
  290.  
  291.         //Set the colour to black (finished), increase the time and set the finishing time
  292.         $nodeU->colour        = Jadva_Tc_Node::COLOUR_BLACK;
  293.         $nodeU->finishingTime = ++$this->_dfsTime;
  294.     }
  295.     //------------------------------------------------
  296.     /**
  297.      * Cleans up the references to the child objects
  298.      *
  299.      * @return void 
  300.      */
  301.     protected function _innerDelete()
  302.     {
  303.         foreach($this->_nodeList as $node) {
  304.             $node->delete();
  305.         }
  306.  
  307.         //All the edges are already dead
  308.  
  309.         unset($this->_nodeList);
  310.         unset($this->_edgeList);
  311.     }
  312.     //------------------------------------------------
  313. }
