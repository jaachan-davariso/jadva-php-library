  286.     protected function _runPage()
  287.     {
  288.         $success = FALSE;
  289.         switch( $this->_page ) {
  290.         case 1: $success = $this->_runPageTestPhpVars(); break;
  291.         case 2: $success = $this->_runPageTestIncludes(); break;
  292.         case 3: die('Not build yet');
  293.         default:
  294.             /** @see Jadva_Installer_Exception */
  295.             require_once 'Jadva/Installer/Exception.php';
  296.             throw new Jadva_Installer_Exception('Invalid page: ' . $this->_page);
  297.         }
  298.  
  299.         if( $success ) {
  300.             $this->_resultsCookie['latestFinishedPage']++;
  301.         }
  302.     }
  303.     //------------------------------------------------
  304.     /**
  305.      * Runs the PHP variable tests
  306.      *
  307.      * @return  boolean  TRUE if the required PHP tests success, FALSE otherwise
  308.      */
  309.     protected function _runPageTestPhpVars()
  310.     {
  311.         $tableRequired = array();
  312.         $tableAdvised  = array();
  313.  
  314.         $result = TRUE;
  315.         foreach($this->_testPhpVarList as $varTest) {
  316.             $testResult = $varTest->test();
  317.  
  318.             $tableRow = array($varTest->getVarName(), $varTest->renderExpectedValue(), $varTest->renderActualValue(), $testResult ? 'success' :'failed');
  319.  
  320.             if( $varTest->getRequired() ) {
  321.                 $result &= $testResult;
  322.  
  323.                 $tableRequired[] = $tableRow;
  324.             } else {
  325.                 $tableAdvised[] = $tableRow;
  326.             }
  327.         }
  328.  
  329.         if( NULL !== $this->_miniumPhpVersion ) {
  330.             $versionTest = version_compare(PHP_VERSION, $this->_miniumPhpVersion, '>=');
  331.  
  332.             $result &= $versionTest;
  333.         } else {
  334.             $versionTest = NULL;
  335.         }
  336.  
  337.         //Output
  338.         $header = array(array('PHP Variable', 'Expected value', 'Actual value', 'Result'));
  339.  
  340.         $this->_outputHtmlHeader();
  341.         $this->_outputHtml('<h1>Tests for PHP Values:</h1>');
  342.  
  343.         if( NULL !== $versionTest ) {
  344.             $this->_outputHtml('<h2>PHP Version:</h2>');
  345.  
  346.             $out = 'Your PHP version is ' . PHP_VERSION . '.';
  347.             if( $versionTest ) {
  348.                 $out .= ' That is sufficient to match the requirements of this application (' . $this->_miniumPhpVersion . ' or higher).';
  349.             } else {
  350.                 $out .= ' This application requires at least ' . $this->_miniumPhpVersion;
  351.             }
  352.  
  353.             $this->_outputHtml('<p class="' . ($versionTest ? 'success' : 'failed') . '">' . $out . '</p>');
  354.         }
  355.  
  356.         if( 0 < count($tableRequired) ) {
  357.             $this->_outputHtml('<h2>Required variable values:</h2>');
  358.             $this->_outputHtmlTable($tableRequired, $header);
  359.         }
  360.  
  361.  
  362.         if( 0 < count($tableAdvised) ) {
  363.             $this->_outputHtml('<h2>Advised variable values:</h2>');
  364.             $this->_outputHtmlTable($tableAdvised, $header);
  365.         }
  366.  
  367.         $this->_outputHtml('<h2>Results:</h2>');
  368.         if( $result ) {
  369.             $this->_outputHtml('<p>All required variable tests passed!');
  370.             if( 0 < count($tableAdvised) ) {
  371.                 $this->_outputHtml(' Some advised variable tests did not pass. It is advisable to update your PHP configuration so all advised variable values match as well.');
  372.             }
  373.  
  374.             $this->_outputHtml(' You can continue to <a href="./index.php">the next step</a><p>');
  375.         } else {
  376.             $this->_outputHtml('<p>Some required tests did not pass. Please update your PHP configuration and <a href="./index.php">try again</a><p>');
  377.         }
  378.  
  379.         $this->_outputHtmlFooter();
  380.  
  381.         return $result;
  382.     }
  383.     //------------------------------------------------
  384.     /**
  385.      * Runs the include tests
  386.      *
  387.      * @return  boolean  TRUE if the include tests success, FALSE otherwise
  388.      */
  389.     protected function _runPageTestIncludes()
  390.     {
  391.         $errorList = array();
  392.  
  393.         if( $this->_allowUserIncludes ) {
  394.             if( !array_key_exists('includeList', $this->_resultsCookie) ) {
  395.                 $this->_resultsCookie['includeList'] = array();
  396.  
  397.                 //Since this class is running, the directory below is in the include path. However, this could
  398.                 //be because of the installer script. It could be that the application requires our libray and
  399.                 //that it's not default in the path. The only way to be really sure that such an application
  400.                 // won't boop up, is to add ourselves always to the include path.
  401.                 $this->_resultsCookie['includeList'][] = dirname(dirname(__FILE__)) . DIRECTORY_SEPARATOR;
  402.             }
  403.  
  404.             if( !empty($_POST['additional_include_paths']) ) {
  405.                 $list = explode(PHP_EOL, $_POST['additional_include_paths']);
  406.                 foreach($list as $path) {
  407.                     $path = trim($path);
  408.                     $path = str_replace(array('/', '\\'), DIRECTORY_SEPARATOR, $path);
  409.                     $path = rtrim($path, DIRECTORY_SEPARATOR) . DIRECTORY_SEPARATOR;
  410.  
  411.                     if( is_dir($path) && is_readable($path) ) {
  412.                         if( !in_array($path, $this->_resultsCookie['includeList']) ) {
  413.                             $this->_resultsCookie['includeList'][] = $path;
  414.                         }
  415.                     } else {
  416.                         $errorList[] = 'Could not read "' . $path . '"';
  417.                     }
  418.                 }
  419.             }
  420.  
  421.  
  422.             //Add the already set include directories to prevent false negatives
  423.             $currentPathList = explode(PATH_SEPARATOR, get_include_path());
  424.             foreach($this->_resultsCookie['includeList'] as $includeDirectory) {
  425.                 if( !in_array($includeDirectory, $currentPathList) ) {
  426.                     set_include_path($includeDirectory . PATH_SEPARATOR . get_include_path());
  427.                 }
  428.             }
  429.         }
  430.  
  431.  
  432.         //Output and do tests
  433.         $this->_outputHtmlHeader();
  434.  
  435.         if( 0 < count($errorList) ) {
  436.             $this->_outputHtml('<h2>Some includes could not be added:</h2>');
  437.             $this->_outputHtml('<ul>');
  438.             foreach($errorList as $error) {
  439.                 $this->_outputHtml('<li>' . $error . '</li>');
  440.             }
  441.             $this->_outputHtml('</ul>');
  442.         }
  443.  
  444.         $this->_outputHtml('<h1>Tests for includes:</h1>');
