  346.     /** Implements Jadva_Installer_Database_Abstract::_parseDatabaseScript */
  347.     protected function _parseDatabaseScript($scriptContents)
  348.     {

  361.             if( empty($line) ) {

  364.  
  365.             $comment = '';

  372.             if( '#' === $line[0] ) {

  374.                 $comment = trim(substr($line, 2));
  375.             }
  376.  
  377.             if( !empty($comment) ) {
  378.                 if( substr($comment, 0, 9) === 'REQUIRES:' ) {
  379.                     $requirement = substr($comment, 9);
  380.                     if( FALSE === strpos($requirement, ',') ) {
  381.                         throw new Exception('Error at line ' . $lineNumber . ': Invalid requirement');
  382.                     }
  383.  
  384.                     list($scriptName, $scriptVersion) = explode(',', $requirement);
  385.                     $return['requirement_list'][] = array(
  386.                         'scriptName'    => $scriptName,

  390.                 continue;
  391.             }
  392.  
  393.             if( substr($line, 0, 9) === 'DELIMITER' ) {

  396.             }
  397.  
  398.             if( FALSE !== ($delPos = strpos($line, $delimeter)) ) {
  399.                 $current_query .= substr($line, 0, $delPos);
  400.                 $return['query_list'][] = $current_query;
  401.  
  402.                 $current_query = trim(substr($line, $delPos + 1)) . PHP_EOL;;
  403.                 continue;
  404.             }
  405.  
  406.             $current_query .= $line . PHP_EOL;
  407.         }
  408.  
  409.         return $return;
  410.     }
