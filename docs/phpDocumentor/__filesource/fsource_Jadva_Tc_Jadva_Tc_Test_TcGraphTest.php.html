  131.     public function topologicalSortTests()
  132.     {
  133.         $graph = new Jadva_Tc_Graph;
  134.  
  135.         $node[0]       = $graph->addNode();
  136.         $identity[0]   = $node[0]->getIdentity();
  137.         $list          = $graph->getNodesAsTopologicalSort();
  138.         $this->_assertTrue(is_array($list),   '"Retrieve topological sort", returned variable is an array', FALSE);
  139.         $this->_assertTrue(count($list) == 1, '"Retrieve topological sort", returned 1 node', TRUE);
  140.  
  141.         $identityMatch = array_key_exists($identity[0], $list);
  142.         $this->_assertTrue($identityMatch,    '"Retrieve topological sort", returned the right identity', TRUE);
  143.  
  144.         $edge[0][0]    = $graph->addEdge($node[0], $node[0]);
  145.         try {
  146.             $list      = $graph->getNodesAsTopologicalSort();
  147.             $success   = FALSE;
  148.         } catch( Jadva_Tc_Graph_Exception $e ) {
  149.             $success   = TRUE;
  150.         }
  151.         $this->_assertTrue($success, '"Retrieve topological sort", retrieving a cyclic graph', TRUE);
  152.         $graph->removeEdge($edge[0][0]);
  153.  
  154.         $node[1]       = $graph->addNode();
  155.         $identity[1]   = $node[1]->getIdentity();
  156.         $list          = $graph->getNodesAsTopologicalSort();
  157.         $this->_assertTrue(count($list) == 2, '"Retrieve topological sort", returned 2 nodes', TRUE);
  158.  
  159.         $identityMatch = array_key_exists($identity[0], $list) && array_key_exists($identity[1], $list);
  160.         $this->_assertTrue($identityMatch   , '"Retrieve topological sort", returned the right identity', TRUE);
  161.  
  162.         $edge[0][1]    = $graph->addEdge($node[0], $node[1]);
  163.         $list          = $graph->getNodesAsTopologicalSort();
  164.         $identityOrder = array($identity[1], $identity[0]);
  165.  
  166.         $this->_assertIdentityListMatches($list, $identityOrder, '"Retrieve topological sort", with 2 linked nodes', TRUE);
  167.  
  168.         $node[2]       = $graph->addNode();
  169.         $identity[2]   = $node[2]->getIdentity();
  170.         $edge[2][0]    = $graph->addEdge($node[0], $node[1]);
  171.         $list          = $graph->getNodesAsTopologicalSort();
  172.         $identityOrder = array($identity[1], $identity[0], $identity[2]);
  173.         $this->_assertIdentityListMatches($list, $identityOrder, '"Retrieve topological sort", with 3 linked nodes', TRUE);
  174.  
  175.         $graph->delete();
  176.         unset($graph);
  177.     }
